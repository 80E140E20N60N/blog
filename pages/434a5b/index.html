# Vue实现动态锚点

实现实现动态锚点 (opens new window)

<template>
  <div class="page">
    <div class="main">
      <div class="side">
        <div class="anchor">
          <div
            v-for="(item, index) in anchorArr"
            :key="index"
            class="anchorItem"
            :class="{ active: activeBtn == index }"
            @click="goAnchor('#anchor-' + index, index)"
          >
            {{ item }}
          </div>
        </div>
      </div>
      <div class="content">
        <div id="#anchor-0" class="contentItem">锚点1内容</div>
        <div id="#anchor-1" class="contentItem">锚点2内容</div>
        <div id="#anchor-2" class="contentItem">锚点3内容</div>
        <div id="#anchor-3" class="contentItem">锚点4内容</div>
        <div id="#anchor-4" class="contentItem">锚点5内容</div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: "App",
  data() {
    return {
      anchorArr: ["锚点1", "锚点2", "锚点3", "锚点4", "锚点5"],
      activeBtn: 0,
      contentArr: [
        "#anchor-0",
        "#anchor-1",
        "#anchor-2",
        "#anchor-3",
        "#anchor-4",
      ],
      distanceArr: [],
      distanceFlag: false,
      distanceTimer: 0,
    };
  },
  mounted() {
    this.$nextTick(function () {
      window.addEventListener("scroll", this.onScroll);
    });
  },
  methods: {
    goAnchor(selector, index) {
      clearTimeout(this.distanceTimer);
      this.distanceFlag = true;
      //切换锚点列表高亮
      this.activeBtn = index;
      // 点击锚点列表项，右侧页面滚动到相应内容
      document.getElementById(selector).scrollIntoView({
        behavior: "smooth",
        block: "center",
        inline: "nearest",
      });
      this.distanceTimer = setTimeout(() => {
        this.distanceFlag = false;
      }, 1000);
    },
    /**
     * 获取右侧各内容块尺寸
     */
    getBoxSize() {
      this.contentArr.forEach((item) => {
        const size =
          document.getElementById(item).offsetTop +
          document.getElementById(item).clientHeight -
          200;
        this.distanceArr.push(size);
      });
    },
    onScroll() {
      if (!this.distanceFlag) {
        const scrolled =
          document.documentElement.scrollTop ||
          window.pageYOffset ||
          document.body.scrollTop;

        this.getBoxSize();

        if (scrolled < this.distanceArr[0]) {
          this.activeBtn = 0;
        } else if (scrolled < this.distanceArr[1]) {
          this.activeBtn = 1;
        } else if (scrolled < this.distanceArr[2]) {
          this.activeBtn = 2;
        } else if (scrolled < this.distanceArr[3]) {
          this.activeBtn = 3;
        } else {
          this.activeBtn = 4;
        }
      }
    },
  },
};
</script>
