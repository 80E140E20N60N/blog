# 使用Echarts地图

echarts内置了世界地图、中国及各个省市自治区地图数据，可通过标准GeoJson扩展地图类型

# 地图点击事件

<template>
  <div>
    <div ref="map" class="map" />
  </div>
</template>

<script>
import taizhou from '@/assets/geojson/taizhou'
import Echarts from 'echarts'
Echarts.registerMap('taizhou', taizhou)
export default {
  name: 'mapDemo',
  mounted() {
    this.drawMap()
  }, 
  methods: { 
    // Eachart参数字体大小转换
    transformSize(res) {
      const docEl = document.documentElement
      const clientWidth = window.innerWidth || docEl.clientWidth || document.body.clientWidth
      if (!clientWidth) return
      const fontSize = 100 * (clientWidth / 1920)
      return parseInt(res * fontSize)
    },
    drawMap() {
      const mapEchart = Echarts.init(this.$refs.map)
      const option = {
        geo: {
          map: 'taizhou',
          aspectScale: 0.95,
          zoom: 0.1, // 当前视角的缩放比例
          layoutCenter: ['50%', '50%'], // 地图位置
          layoutSize: '100%',
          itemStyle: {
            normal: {
              shadowColor: '#007FFF',
              shadowOffsetX: 7,
              shadowOffsetY: 7,
              opacity: 0.5
            },
            emphasis: { areaColor: '#007FFF' }
          }
        },
        series: [
          {
            type: 'map',
            mapType: 'taizhou',
            aspectScale: 0.85,
            layoutCenter: ['50%', '50%'], // 地图位置
            layoutSize: '100%',
            zoom: 1.6, // 当前视角的缩放比例
            roam: false, // 是否开启平游或缩放
            itemStyle: {
              normal: {
                areaColor: '#073B86',
                borderColor: '#ABD5FF',
                borderWidth: 1,
                label: {
                  show: true,
                  color: '#A9C7E4',
                  fontSize: this.transformSize(0.18)
                },
                emphasis: {
                  label: { show: true }
                }
              }
            },
            emphasis: { // 高亮状态
              show: false,
              itemStyle: { areaColor: '#0085FF' },
              label: {
                show: true,
                position: 'top',
                color: '#FFFFFF',
                backgroundColor: 'rgba(3, 16, 41, 0.72)',
                padding: [10, 16],
                borderRadius: 6,
                borderColor: '#0062FF',
                borderWidth: 1,
                borderType: 'solid',
                shadowOffsetX: 0,
                shadowOffsetY: 0,
                shadowBlur: 16,
                shadowColor: '#005EFF'
              }
            }
          }
        ]
      }
      mapEchart.setOption(option)       
      // 地图点击当前区域块高亮
      mapEchart.on('click', (e) => { 
        mapEchart.dispatchAction({
          type: 'downplay',
          seriesIndex: 0
        })
        mapEchart.dispatchAction({
          type: 'highlight',
          seriesIndex: 0,
          dataIndex: e.dataIndex
        })          
      })
  }
}
</script>

<style lang="scss" scoped>
    .map{
        width: 5rem;
        height: 5rem;
    }
</style>

# 地图区域划分调整

可以通过编辑geojson地图数据 (opens new window)来自定义地图区域

1、将需要调整区域的JSON文件导入，网站自动显示该区域地图
2、点击区域块后，有tooltip弹窗，点击右下角“delete feature”可以删除区域块
3、使用多边形工具圈出所需区域，点击绘制的区域块，添加属性“name”
4、制作完成后将右侧的json复制压缩即可使用
