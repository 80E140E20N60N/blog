# 适配器模式

  1. 场景：方法名字改变时，面临接口不通用的问题
  2. 目的：写适配器进行代替替换
  3. 例子
// 数据和参数适配
// 例：后端数据结构不适合时可以考虑数据适配
const obj={
    t1:...,
    t2:...
}

# 装饰者模式

  1. 场景：方法的作用发生改变时，当方法需要扩展时但是不好直接去修改方法
  2. 目的：不重写方法的扩展方法
  3. 例子
// 1、新建自己方法
// 2、内部调用老方法
// 3、执行自己的方法

// 例：给dom已经有的时间，增加操作
var _old = dom.onclick
dom.onclick = function(){
    _old();
    fn();
}

// 使用工厂模式升级
var decorator =function(dom,fn){
    if(typeof dom.onclick == 'function'){
        var _old = dom.onclick
        dom.onclick = function(){
            _old();
            fn();
        }
    }
}

decorator(deleteBtn,function(){
    console.log('删除')
})
